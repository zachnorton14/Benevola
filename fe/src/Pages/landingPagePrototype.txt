import { useNavigate } from 'react-router-dom';
import { useState } from 'react';
import '../App.css';
import NavBar from "../Components/NavBar"
// alternate arrow icon import EastIcon from '@mui/icons-material/East';
import ArrowForwardIcon from '@mui/icons-material/ArrowForward';

function LandingPage() {
  const navigate = useNavigate();

  // used for hovering over a button
  const [eventHovered, setEventHovered] = useState(false);
  const [orgHovered, setOrgHovered] = useState(false);

  const handleEvent = () => {
    navigate('/opp-list');
  };

  const handleOrg = () => {
    navigate('/create-event');
  };

  return (
    
    <div> 
      
      <head>
        <title> Benevola Landing page</title>
      </head>

      <div className="landing-navbar">
        <NavBar/>
      </div>

      {/** Background Image */}
      <div className="landing-page-image-container">
        <div className="landing-page-title-container">
          <div className="landing-page-title-text">
            Benevola
          </div>
          <div className="landing-page-title-description-text">
            Connecting Volunteers with Organizations accross the world.
          </div>
        </div>
      </div>

      <div className="landing-page-organization-container">
        <div className="landing-page-organization-text">
          Looking to find a place to volunteer?
        </div>

        <div className="landing-page-organization-text-secondary">
          Find events made for you.
        </div>
        
        <button  
          onClick={handleEvent}
          onMouseEnter={() => setEventHovered(true)}
          onMouseLeave={() => setEventHovered(false)}
          className="landing-page-event-button">
          Discover
          {eventHovered ? <> <div className="landing-page-event-icon-container">
            <ArrowForwardIcon fontSize="large"/>
          </div> </> : <></>} 
        </button>
      </div>

  

      <div className='landing-page-event-container'>
         <div className="landing-page-organization-container">
          <div className="landing-page-organization-text">
            
            Need Volunteers for your event?
          </div>

          <div className="landing-page-organization-text-secondary">
            Register your event with Benevola.
          </div>
          
          <button  
            onClick={handleOrg}
            onMouseEnter={() => setOrgHovered(true)}
            onMouseLeave={() => setOrgHovered(false)}
            className="landing-page-organization-button">
            Get Started
            {orgHovered ? <> <div className="landing-page-organization-icon-container">
              <ArrowForwardIcon fontSize="large"/>
            </div> </> : <></>} 
          </button>
        </div>

      </div>


    </div>
  );
}

export default LandingPage;
